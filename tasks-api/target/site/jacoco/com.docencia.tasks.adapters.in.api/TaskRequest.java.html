package com.docencia.tasks.adapters.in.api;

/**
 * DTO (Data Transfer Object) que representa la solicitud de creación o
 * actualización de una tarea.
 */
public class TaskRequest {

  /** Título de la tarea. */
  private String title;

  /** Descripción detallada de la tarea. */
  private String description;

  /**
   * Estado de completitud de la tarea. Puede ser nulo en actualizaciones
   * parciales.
   */
  private Boolean completed;

  public TaskRequest() {
  }

  public String getTitle() {
    return title;
  }

  public String getDescription() {
    return description;
  }

  public Boolean getCompleted() {
    return completed;
  }

  public void setTitle(String title) {
    this.title = title;
  }

  public void setDescription(String description) {
    this.description = description;
  }

  public void setCompleted(Boolean completed) {
    this.completed = completed;
  }
}
